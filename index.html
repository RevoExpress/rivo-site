const express = require('express'); const nodemailer = require('nodemailer'); const cors = require('cors'); require('dotenv').config(); const app = express(); // Middleware app.use(cors()); app.use(express.json()); app.use(express.static('public')); // Configurer Nodemailer const transporter = nodemailer.createTransport({ service: 'gmail', auth: { user: process.env.GMAIL_USER, pass: process.env.GMAIL_PASS } }); // Route GET - Afficher le site app.get('/', (req, res) => { res.sendFile(__dirname + '/public/index.html'); }); // Route POST - Envoyer commande par email app.post('/api/send-order', async (req, res) => { try { const { nom, telephone, adressePickup, adresseLivraison, details } = req.body; // Validation if (!nom || !telephone || !adressePickup || !adresseLivraison || !details) { return res.status(400).json({ error: 'Tous les champs sont obligatoires' }); } // HTML Email const htmlContent = `

🚀 NOUVELLE COMMANDE

Client:

Nom: ${nom}

Téléphone: ${telephone}

Adresses:

📍 Pickup: ${adressePickup}

📍 Livraison: ${adresseLivraison}

Détails:

${details}

${new Date().toLocaleString('fr-FR')}

`; // Envoyer l'email await transporter.sendMail({ from: process.env.GMAIL_USER, to: process.env.GMAIL_USER, subject: `📦 Commande REVO EXPRESS - ${nom}`, html: htmlContent }); res.json({ success: true, message: '✅ Commande envoyée!' }); } catch (error) { console.error('Erreur:', error); res.status(500).json({ error: 'Erreur lors de l\'envoi' }); } }); // Démarrer const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`🚀 Serveur sur http://localhost:${PORT}`); });
